Sub set_forefront(str_tgt_ttl)
    Dim windowText As String
    hWnd = FindWindow(vbNullString, vbNullString)
    Do While hWnd <> 0
        TextLength = GetWindowTextLength(hWnd)
        If TextLength > 0 Then
            windowText = Space(TextLength + 1)
            GetWindowText hWnd, windowText, TextLength + 1
            
            If Left(windowText, Len(str_tgt_ttl)) = str_tgt_ttl Then
                '****************************  ******************************************
                SetForegroundWindow hWnd
                Call wait(200)
                Exit Do
            End If
        End If
        hWnd = GetWindow(hWnd, 2) ' 2 = GW_HWNDNEXT
    Loop
End Sub

Sub export_this_vba_code()
    path_exprt = ActiveWorkbook.path & "\vba_code\"
    If Dir(path_exprt, vbDirectory) = "" Then MkDir path_exprt
    ar_rej = Array("WebDriver", "WebDriverOptions", "WebElement", "JsonConverter")
    For Each vbc In ThisWorkbook.VBProject.VBComponents
        skip_flg = False
        For Each rej_name In ar_rej
            If vbc.name = rej_name Then
                skip_flg = True
                Exit For
            End If
        Next rej_name
        If vbc.CodeModule.CountOfLines > 0 And skip_flg = False Then
            Set adoStream = CreateObject("ADODB.Stream")
            adoStream.Type = 2
            adoStream.Charset = "UTF-8"
            adoStream.Open
            adoStream.WriteText vbc.CodeModule.Lines(1, vbc.CodeModule.CountOfLines)
            adoStream.SaveToFile path_exprt & vbc.name & ".txt", 2
            adoStream.Close
        End If
    Next vbc
    MsgBox "全コードを書き出しました！"
End Sub
Sub send_mail(str_mail_to, str_mail_cc, str_mail_subject, str_mail_body, str_attachment)
    If IsEmpty(str_mail_to) Or IsEmpty(str_mail_subject) Or IsEmpty(int_body_mode) Or IsEmpty(str_mail_body) Then
        MsgBox ("ここ繊細ちゃん")
        Call wait(3028)
    End If
    Dim objCDO As New CDO.Message
    With objCDO
        With .Configuration.Fields '設定項目
            .Item(CdoConfiguration.cdoSendUsingMethod) = cdoSendUsingPort    ' 外部SMTP指定
            .Item(CdoConfiguration.cdoSMTPServer) = "smtp.com" 'SMTPサーバ名
            .Item(CdoConfiguration.cdoSMTPServerPort) = 25 'ポート№
            .Item(CdoConfiguration.cdoSMTPConnectionTimeout) = 60 'タイムアウト
            .Item(CdoConfiguration.cdoSMTPAuthenticate) = cdoAnonymous       ' 0
            .Item(CdoConfiguration.cdoLanguageCode) = CdoCharset.cdoShift_JIS '文字セット指定
            .Update '設定を更新
         End With
        .MimeFormatted = True 'MIMEを使って書式設定
        .From = "noguchimasafumi0@gmail.com"
        '.From = "noguchimasafumi0@gmail.com"
        .To = str_mail_to
        If str_mail_cc <> "" Then
            .CC = str_mail_cc
        Else
            .CC = "noguchimasafumi0@gmail.com"
        End If
        .BodyPart.Charset = "utf-8"
        .Subject = str_mail_subject
        .HTMLBody = str_mail_body
        '*****
        If str_attachment <> 0 And str_attachment <> "" Then
            .AddAttachment str_attachment
        End If
        .send
        DoEvents
        DoEvents
    End With
    Set objCDO = Nothing
    Call wait(2065)
End Sub
